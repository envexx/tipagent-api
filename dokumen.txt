1. Project Overview
1.1 Ringkasan
TipAgent adalah autonomous AI agent yang mengirim USDT tip secara otomatis kepada kontributor berdasarkan event dari GitHub, Discord, dan custom webhook. Agent menggunakan Tether WDK untuk self-custodial wallet, Google Gemini 2.0 Flash sebagai reasoning engine, dan Aave V3 untuk yield dari USDT idle di treasury.

Stack Decisions — v2 Final
LLM: Google Gemini 2.0 Flash — structured JSON output native, cost-efficient.
Storage: Cloudflare D1 SAJA — rate limit dan cooldown pindah ke tabel D1 rate_limits. Tidak ada KV.
Queue: Cloudflare Queues native — zero external dependency, retry built-in.
Aave V3 Yield: WAJIB ADA — ini differentiator utama vs semua kompetitor.
Backend: Hono.js ke CF Workers. Frontend: React ke CF Pages.

1.3 Alur End-to-End
•	GitHub/Discord/Webhook kirim POST ke Hono endpoint
•	HMAC diverifikasi, event dinormalisasi ke TipEvent
•	Event di-push ke Cloudflare Queue (webhook return 200 instan)
•	Queue consumer: Rule Engine (D1) → Gemini evaluate → WDK transfer USDT
•	CF cron setiap 30 menit: cek liquid balance → deposit excess ke Aave V3
•	Post-tip: auto-comment di GitHub/Discord dengan tx hash Basescan link

4.2 apps/api/src/db/queries.ts
import type { D1Database } from '@cloudflare/workers-types'
import type { TipEvent }   from '@tipagent/shared'

// ── Rate limits (D1, bukan KV) ────────────────────────────────────────────
export async function getRateLimit(db:D1Database, key:string): Promise<string|null> {
  const r = await db.prepare('SELECT value FROM rate_limits WHERE key=? AND expires_at>?')
    .bind(key, Date.now()).first<{value:string}>()
  return r?.value ?? null
}

export async function setRateLimit(db:D1Database,key:string,value:string,ttlMs:number) {
  const exp = Date.now() + ttlMs
  await db.prepare('INSERT INTO rate_limits(key,value,expires_at) VALUES(?,?,?) ON CONFLICT(key) DO UPDATE SET value=excluded.value,expires_at=excluded.expires_at')
    .bind(key,value,exp).run()
}

export async function cleanExpiredRateLimits(db:D1Database) {
  await db.prepare('DELETE FROM rate_limits WHERE expires_at<?').bind(Date.now()).run()
}

// ── Tip history ───────────────────────────────────────────────────────────
export async function saveTipHistory(db:D1Database, a:{
  eventId:string; event:TipEvent; recipientAddr:string
  amountUsdt:number; reasoning:string; status?:string
}) {
  await db.prepare('INSERT INTO tip_history(event_id,source,event_type,recipient_id,recipient_addr,amount_usdt,reasoning,chain,status,created_at) VALUES(?,?,?,?,?,?,?,?,?,?)')
    .bind(a.eventId,a.event.source,a.event.eventType,a.event.recipient.id,
         a.recipientAddr,a.amountUsdt.toString(),a.reasoning,"base",a.status??"pending",Date.now()).run()
}

export async function updateTipStatus(db:D1Database,eventId:string,status:"confirmed"|"failed",txHash?:string,err?:string) {
  await db.prepare('UPDATE tip_history SET status=?,tx_hash=?,error_msg=?,confirmed_at=? WHERE event_id=?')
    .bind(status,txHash??null,err??null,Date.now(),eventId).run()
}

export async function saveAuditLog(db:D1Database,a:{eventId:string;event:TipEvent;result:string;reason?:string}) {
  await db.prepare('INSERT INTO event_audit(event_id,source,event_type,result,reject_reason,raw_payload,created_at) VALUES(?,?,?,?,?,?,?)')
    .bind(a.eventId,a.event.source,a.event.eventType,a.result,a.reason??null,JSON.stringify(a.event.rawPayload),Date.now()).run()
}

export async function getRecentTips(db:D1Database,limit=20) {
  return db.prepare('SELECT * FROM tip_history ORDER BY created_at DESC LIMIT ?').bind(limit).all()
}

export async function getTipStats(db:D1Database) {
  const tot = await db.prepare('SELECT COUNT(*) as c, COALESCE(SUM(CAST(amount_usdt AS REAL)),0) as s FROM tip_history WHERE status="confirmed"').first<{c:number;s:number}>()
  const today = new Date().toISOString().split("T")[0]
  const tod = await db.prepare('SELECT COUNT(*) as c FROM tip_history WHERE status="confirmed" AND created_at>?').bind(new Date(today).getTime()).first<{c:number}>()
  const usr = await db.prepare('SELECT COUNT(DISTINCT recipient_id) as c FROM tip_history WHERE status="confirmed"').first<{c:number}>()
  return{totalTips:tot?.c??0,totalUsdt:tot?.s??0,todayTips:tod?.c??0,activeUsers:usr?.c??0}
}

export async function saveTreasurySnapshot(db:D1Database,a:{liquidUsdt:number;aaveUsdt:number;aaveApy?:string}) {
  await db.prepare('INSERT INTO treasury_snapshots(liquid_usdt,aave_usdt,total_usdt,aave_apy,snapshot_at) VALUES(?,?,?,?,?)')
    .bind(a.liquidUsdt.toString(),a.aaveUsdt.toString(),(a.liquidUsdt+a.aaveUsdt).toString(),a.aaveApy??null,Date.now()).run()
}

6.2 triggers/verifier.ts
export async function verifyGitHubSignature(payload:string,sig:string,secret:string):Promise<boolean>{
  const enc=new TextEncoder()
  const key=await crypto.subtle.importKey("raw",enc.encode(secret),{name:"HMAC",hash:"SHA-256"},false,["sign"])
  const mac=await crypto.subtle.sign("HMAC",key,enc.encode(payload))
  const expected="sha256="+[...new Uint8Array(mac)].map(b=>b.toString(16).padStart(2,"0")).join("")
  return timingSafe(expected,sig)
}
export async function verifyHmac(payload:string,sig:string,secret:string):Promise<boolean>{
  const enc=new TextEncoder()
  const key=await crypto.subtle.importKey("raw",enc.encode(secret),{name:"HMAC",hash:"SHA-256"},false,["sign"])
  const mac=await crypto.subtle.sign("HMAC",key,enc.encode(payload))
  const computed=btoa(String.fromCharCode(...new Uint8Array(mac)))
  return timingSafe(computed,sig)
}
function timingSafe(a:string,b:string):boolean{
  if(a.length!==b.length)return false
  let r=0; for(let i=0;i<a.length;i++)r|=a.charCodeAt(i)^b.charCodeAt(i)
  return r===0
}

6.3 triggers/normalizer.ts
import type { TipEvent } from '@tipagent/shared'
export function normalizeGitHubPR(p:any):TipEvent|null{
  if(p.action!=="closed"||!p.pull_request?.merged)return null
  const pr=p.pull_request
  return{id:crypto.randomUUID(),source:"github",eventType:"pr_merged",
    actor:{id:pr.merged_by?.login??pr.user.login},
    recipient:{id:pr.user.login,displayName:pr.user.login},
    context:{repoName:p.repository.full_name,prNumber:pr.number,prTitle:pr.title,
             prAdditions:pr.additions,prDeletions:pr.deletions},
    timestamp:Date.now(),rawPayload:p}
}
export function normalizeGitHubIssue(p:any):TipEvent|null{
  if(p.action!=="closed"||!p.issue)return null
  const i=p.issue
  return{id:crypto.randomUUID(),source:"github",eventType:"issue_closed",
    actor:{id:p.sender.login},
    recipient:{id:i.user.login,displayName:i.user.login},
    context:{repoName:p.repository.full_name,issueNumber:i.number,
             issueLabels:i.labels?.map((l:any)=>l.name)??[]},
    timestamp:Date.now(),rawPayload:p}
}
export function normalizeDiscordReaction(p:any):TipEvent|null{
  return{id:crypto.randomUUID(),source:"discord",eventType:"discord_reaction",
    actor:{id:"system"},recipient:{id:p.message_author_id},
    context:{discordChannelId:p.channel_id,discordMessageId:p.message_id,reactionCount:p.reaction_count},
    timestamp:Date.now(),rawPayload:p}
}

6.4 routes/webhooks/github.ts
import { Hono } from 'hono'
import type { Env } from '../../index'
import { verifyGitHubSignature } from '../../triggers/verifier'
import { normalizeGitHubPR, normalizeGitHubIssue } from '../../triggers/normalizer'
export const githubRoute=new Hono<{Bindings:Env}>()
githubRoute.post("/",async(c)=>{
  const sig=c.req.header("X-Hub-Signature-256")??""
  const body=await c.req.text()
  if(!await verifyGitHubSignature(body,sig,c.env.GITHUB_WEBHOOK_SECRET))
    return c.json({error:"Invalid signature"},401)
  const payload=JSON.parse(body)
  const event=c.req.header("X-GitHub-Event")
  const tipEvent=event==="pull_request"?normalizeGitHubPR(payload)
    :event==="issues"?normalizeGitHubIssue(payload):null
  if(!tipEvent)return c.json({ok:true,skipped:true})
  await c.env.TIP_QUEUE.send({type:"tip_event",event:tipEvent})
  return c.json({ok:true,eventId:tipEvent.id})
})

import { GoogleGenerativeAI, SchemaType } from '@google/generative-ai'
import type { TipEvent, TipDecision }     from '@tipagent/shared'
import type { RuleResult }                from '../engine/types'

const SCHEMA={
  type:SchemaType.OBJECT,
  properties:{
    allowed:     {type:SchemaType.BOOLEAN,description:"Whether to send the tip"},
    amountUsdt:  {type:SchemaType.NUMBER, description:"USDT amount"},
    reasoning:   {type:SchemaType.STRING, description:"Why this amount"},
    rejectReason:{type:SchemaType.STRING, description:"Reason if not allowed, else empty string"},
  },
  required:["allowed","amountUsdt","reasoning"],
}

export async function evaluateWithGemini(event:TipEvent,rule:RuleResult,apiKey:string):Promise<TipDecision>{
  const genAI=new GoogleGenerativeAI(apiKey)
  const model=genAI.getGenerativeModel({
    model:"gemini-2.0-flash",
    generationConfig:{responseMimeType:"application/json",responseSchema:SCHEMA,temperature:0.2,maxOutputTokens:256},
    systemInstruction:
      `You are TipAgent, an autonomous USDT tipping bot. Return ONLY valid JSON. `+
      `Amount MUST be between ${rule.minAmount} and ${rule.maxAmount} USDT. `+
      `Suggested range: ${rule.suggestedMin}–${rule.suggestedMax} USDT. `+
      "Low impact → use min range. High impact → use max. Self-tips: always allowed:false.",
  })
  const res=await model.generateContent(buildPrompt(event))
  try{
    const d=JSON.parse(res.response.text()) as TipDecision
    d.amountUsdt=Math.min(Math.max(d.amountUsdt??rule.suggestedMin,rule.minAmount),rule.maxAmount)
    return d
  }catch{
    return{allowed:true,amountUsdt:rule.suggestedMin,reasoning:"Fallback: parse error"}
  }
}

function buildPrompt(e:TipEvent):string{
  if(e.eventType==="pr_merged")
    return `GitHub PR merged. Repo: ${e.context.repoName}. Title: "${e.context.prTitle}". `+
           `Lines added: ${e.context.prAdditions}, removed: ${e.context.prDeletions}. Contributor: ${e.recipient.displayName}.`
  if(e.eventType==="issue_closed")
    return `GitHub Issue closed. Repo: ${e.context.repoName}. Labels: ${e.context.issueLabels?.join(",")??"none"}. Reporter: ${e.recipient.id}.`
  if(e.eventType==="discord_reaction")
    return `Discord message got ${e.context.reactionCount} reactions. Author: ${e.recipient.id}.`
  return `Evaluate: ${JSON.stringify(e.context)}`
}

6.6 engine/ruleEngine.ts — D1-based (No KV)
import type { TipEvent } from '@tipagent/shared'
import type { Env }      from '../index'
import type { RuleResult } from './types'
import { getRateLimit }  from '../db/queries'

export async function evaluateRules(event:TipEvent,env:Env):Promise<RuleResult>{
  const cfg={min:parseFloat(env.TIP_MIN_USDT),max:parseFloat(env.TIP_MAX_USDT),
             cap:parseFloat(env.DAILY_CAP_PER_USER),cd:parseFloat(env.COOLDOWN_HOURS)}
  if(event.actor.id===event.recipient.id)return reject("Self-tip not allowed")
  const cdKey=`cooldown:${event.source}:${event.recipient.id}`
  const last=await getRateLimit(env.DB,cdKey)
  if(last){
    const el=(Date.now()-parseInt(last))/3_600_000
    if(el<cfg.cd)return reject(`Cooldown: ${(cfg.cd-el).toFixed(1)}h remaining`)
  }
  const today=new Date().toISOString().split("T")[0]
  const capKey=`dailycap:${event.recipient.id}:${today}`
  const prev=parseFloat((await getRateLimit(env.DB,capKey))??"0")
  if(prev>=cfg.cap)return reject(`Daily cap: ${prev} USDT sent today`)
  const[sMin,sMax]=tier(event)
  const eff=Math.min(cfg.max,sMax,cfg.cap-prev)
  return{allowed:true,minAmount:cfg.min,maxAmount:eff,suggestedMin:sMin,suggestedMax:Math.min(sMax,eff)}
}

function tier(e:TipEvent):[number,number]{
  switch(e.eventType){
    case"pr_merged":{const a=e.context.prAdditions??0;return a<50?[0.5,2]:a<300?[2,10]:[10,50]}
    case"issue_closed":return (e.context.issueLabels??[]).includes("bug")?[1,5]:[0.5,2]
    case"discord_reaction":return[0.5,1]
    case"discord_command":return[1,10]
    case"discord_thread_resolved":return[1,3]
    default:return[0.5,3]
  }
}
function reject(r:string):RuleResult{return{allowed:false,rejectReason:r,minAmount:0,maxAmount:0,suggestedMin:0,suggestedMax:0}}

6.9 wallet/yieldManager.ts — Aave V3 WAJIB
Aave V3 adalah fitur WAJIB — Jangan hapus atau comment-out
Ini yang membuktikan "economic soundness" ke juri: treasury tidak pernah idle sia-sia.
Jika ada error Aave, log error tapi JANGAN crash agent — tip flow harus tetap jalan.

import{createPublicClient,createWalletClient,http,parseUnits,formatUnits,maxUint256}from 'viem'
import{privateKeyToAccount}from 'viem/accounts'
import{SUPPORTED_CHAINS,ERC20_ABI,AAVE_POOL_ABI,ATOKEN_ABI}from '@tipagent/shared'
import type{Env}from '../index'
import{saveTreasurySnapshot}from '../db/queries'
export class YieldManager{
  private env:Env
  private c=SUPPORTED_CHAINS["base"]
  constructor(env:Env){this.env=env}
  private cl(){
    const acc=privateKeyToAccount(this.env.WDK_PRIVATE_KEY as `0x${string}`)
    const t=http(this.env.BASE_RPC_URL)
    return{pub:createPublicClient({chain:this.c.chain,transport:t}),
           wal:createWalletClient({account:acc,chain:this.c.chain,transport:t}),acc}
  }
  async depositIdleFunds(liquid:number):Promise<string|null>{
    if(this.env.YIELD_ENABLED!=="true")return null
    const reserve=parseFloat(this.env.LIQUID_RESERVE_USDT)
    const excess=liquid-reserve
    if(excess<1)return null
    const{wal,acc}=this.cl()
    await wal.writeContract({address:this.c.usdtAddress,abi:ERC20_ABI,
      functionName:"approve",args:[this.c.aavePool,maxUint256]})
    const hash=await wal.writeContract({address:this.c.aavePool,abi:AAVE_POOL_ABI,
      functionName:"supply",args:[this.c.usdtAddress,parseUnits(excess.toFixed(6),6),acc.address,0]})
    console.log(`[Aave] Deposited ${excess.toFixed(2)} USDT tx:${hash}`)
    return hash
  }
  async withdrawForTip(needed:number):Promise<string>{
    const{wal,acc}=this.cl()
    const hash=await wal.writeContract({address:this.c.aavePool,abi:AAVE_POOL_ABI,
      functionName:"withdraw",args:[this.c.usdtAddress,parseUnits(needed.toFixed(6),6),acc.address]})
    console.log(`[Aave] Withdrew ${needed.toFixed(2)} USDT tx:${hash}`)
    return hash
  }
  async getAaveBalance():Promise<number>{
    const{pub,acc}=this.cl()
    const r=await pub.readContract({address:this.c.aUSDT,abi:ATOKEN_ABI,
      functionName:"balanceOf",args:[acc.address]})
    return parseFloat(formatUnits(r as bigint,6))
  }
  async snapshot(liquid:number):Promise<void>{
    const aave=await this.getAaveBalance().catch(()=>0)
    await saveTreasurySnapshot(this.env.DB,{liquidUsdt:liquid,aaveUsdt:aave})
  }
}

6.10 queue/processor.ts — Full 10-Step Agent Flow
import type{TipEvent}from '@tipagent/shared'
import type{Env}from '../index'
import{evaluateRules}from '../engine/ruleEngine'
import{evaluateWithGemini}from '../agent/brain'
import{WDKManager}from '../wallet/wdkManager'
import{YieldManager}from '../wallet/yieldManager'
import{resolveRecipient}from '../registry/resolver'
import{notifyGitHub}from '../notifier/github'
import{notifyDiscord}from '../notifier/discord'
import{saveAuditLog,saveTipHistory,updateTipStatus,setRateLimit,getRateLimit}from '../db/queries'
export interface QueueMessage{type:"tip_event";event:TipEvent}

export async function processQueueMessage(msg:QueueMessage,env:Env){
  const{event}=msg
  console.log(`[Queue] ${event.id} (${event.eventType})`)
  // Step 1: Audit
  await saveAuditLog(env.DB,{eventId:event.id,event,result:"processing"})
  // Step 2: Resolve wallet
  const rec=await resolveRecipient(event.recipient.id,event.source,env.DB)
  if(!rec){await saveAuditLog(env.DB,{eventId:event.id,event,result:"held",reason:"No wallet"});return}
  event.recipient.walletAddress=rec.wallet_addr
  // Step 3: Rule engine
  const rule=await evaluateRules(event,env)
  if(!rule.allowed){await saveAuditLog(env.DB,{eventId:event.id,event,result:"rejected",reason:rule.rejectReason});return}
  // Step 4: Gemini LLM
  const dec=await evaluateWithGemini(event,rule,env.GEMINI_API_KEY)
  if(!dec.allowed){await saveAuditLog(env.DB,{eventId:event.id,event,result:"rejected",reason:dec.rejectReason});return}
  // Step 5: Save pending
  await saveTipHistory(env.DB,{eventId:event.id,event,recipientAddr:rec.wallet_addr,
    amountUsdt:dec.amountUsdt,reasoning:dec.reasoning,status:"processing"})
  // Step 6: Liquidity + Aave withdraw
  const wdk=new WDKManager(env)
  const bal=await wdk.getUSDTBalance()
  if(bal<dec.amountUsdt){
    try{await new YieldManager(env).withdrawForTip((dec.amountUsdt-bal)*1.1)}
    catch(e){console.error("[Aave] Withdraw failed:",e)}
  }
  // Step 7–10: Transfer + update counters + notify
  try{
    const tx=await wdk.transferUSDT(rec.wallet_addr as `0x${string}`,dec.amountUsdt)
    console.log(`[WDK] ${dec.amountUsdt} USDT → ${rec.wallet_addr} tx:${tx}`)
    const cdKey=`cooldown:${event.source}:${event.recipient.id}`
    const today=new Date().toISOString().split("T")[0]
    const capKey=`dailycap:${event.recipient.id}:${today}`
    const prev=parseFloat((await getRateLimit(env.DB,capKey))??"0")
    await Promise.all([
      setRateLimit(env.DB,cdKey,Date.now().toString(),86_400_000),
      setRateLimit(env.DB,capKey,(prev+dec.amountUsdt).toString(),86_400_000),
      updateTipStatus(env.DB,event.id,"confirmed",tx),
    ])
    if(event.source==="github")await notifyGitHub(event,dec.amountUsdt,tx,env.GITHUB_TOKEN)
    else if(event.source==="discord")await notifyDiscord(event,dec.amountUsdt,tx,env.DISCORD_BOT_TOKEN)
  }catch(e){
    await updateTipStatus(env.DB,event.id,"failed",undefined,String(e))
    throw e
  }
}
